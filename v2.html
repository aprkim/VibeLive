import { VibeLive } from '@vibelive/sdk'
import { VideoRoom } from '@vibelive/react'

const App = () => {
  const { room, participants } = useVibeLive()

  return (
    <VideoRoom
      roomId={room.id}
      onJoin={(user) => console.log('joined', user)}
    >
      {participants.map(p => (
        <VideoTile key={p.id} participant={p} />
      ))}
    </VideoRoom>
  )
}

// No WebRTC configuration needed
// No STUN/TURN servers to manage
// No signaling infrastructure

export default VibeLive.init({
  apiKey: process.env.VIBELIVE_KEY,
  enabled: true
})
